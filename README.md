
## 安装
~~~
npm install --save yu.calculator
~~~
## 使用方法
~~~
var calculator = require('yu.calculator');

// 加
var r1 = calculator.add(0.1, 0.2) // 0.3

// 减
var r2 = calculator.sub(0.51, 0.3) // 0.21

// 乘
var r3 = calculator.mul(7, 0.8) // 5.6

// 除
var r4 = calculator.div(0.3, 0.1) // 3
~~~
